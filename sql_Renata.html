-- Апдейт 2 Ренаты: к стартовой схеме добавлена таблица прогресса обучения.

CREATE TABLE IF NOT EXISTS renata_lessons (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    title TEXT NOT NULL,
    tag TEXT NOT NULL,
    summary TEXT NOT NULL,
    body TEXT NOT NULL,
    sort_order INTEGER NOT NULL DEFAULT 0
);

CREATE TABLE IF NOT EXISTS renata_lesson_steps (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    lesson_id INTEGER NOT NULL,
    step_text TEXT NOT NULL,
    sort_order INTEGER NOT NULL DEFAULT 0,
    FOREIGN KEY (lesson_id) REFERENCES renata_lessons(id)
);

CREATE TABLE IF NOT EXISTS renata_quiz_questions (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    lesson_id INTEGER NOT NULL,
    question_text TEXT NOT NULL,
    sort_order INTEGER NOT NULL DEFAULT 0,
    FOREIGN KEY (lesson_id) REFERENCES renata_lessons(id)
);

CREATE TABLE IF NOT EXISTS renata_quiz_answers (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    question_id INTEGER NOT NULL,
    answer_text TEXT NOT NULL,
    is_correct INTEGER NOT NULL DEFAULT 0,
    sort_order INTEGER NOT NULL DEFAULT 0,
    FOREIGN KEY (question_id) REFERENCES renata_quiz_questions(id)
);

CREATE TABLE IF NOT EXISTS renata_learning_progress (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_key TEXT NOT NULL,
    lesson_id INTEGER NOT NULL,
    completed_at TEXT NOT NULL,
    score INTEGER,
    UNIQUE (user_key, lesson_id),
    FOREIGN KEY (lesson_id) REFERENCES renata_lessons(id)
);
