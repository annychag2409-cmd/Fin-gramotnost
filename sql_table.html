CREATE TABLE IF NOT EXISTS aleria_comparison_criteria (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    criterion TEXT NOT NULL,
    deposit_text TEXT NOT NULL,
    savings_text TEXT NOT NULL,
    investment_text TEXT NOT NULL,
    sort_order INTEGER NOT NULL DEFAULT 0
);

CREATE TABLE IF NOT EXISTS aleria_survey_questions (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    question_text TEXT NOT NULL,
    sort_order INTEGER NOT NULL DEFAULT 0
);

CREATE TABLE IF NOT EXISTS aleria_survey_options (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    question_id INTEGER NOT NULL,
    option_text TEXT NOT NULL,
    result_type TEXT NOT NULL CHECK (result_type IN ('deposit', 'savings', 'investment')),
    sort_order INTEGER NOT NULL DEFAULT 0,
    FOREIGN KEY (question_id) REFERENCES aleria_survey_questions(id)
);

CREATE TABLE IF NOT EXISTS aleria_survey_results (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    result_type TEXT NOT NULL UNIQUE CHECK (result_type IN ('deposit', 'savings', 'investment')),
    title TEXT NOT NULL,
    description TEXT NOT NULL,
    checklist TEXT
);
